"use strict";

const findBestEmployee = function (employees) {
  let max = 0;
  let name;

  const keysEmploy = Object.keys(employees);
  for (const key of keysEmploy) {
    if (max < employees[key]) {
      max = employees[key];
      name = key;
    }
  }
  return console.log(`${name}: ${max}`);
  //   return { name, tasksCompleted: max };
};

console.log(
  findBestEmployee({
    ann: 29,
    david: 35,
    helen: 1,
    lorence: 99,
  })
); // lorence: 99

console.log(
  findBestEmployee({
    poly: 12,
    mango: 17,
    ajax: 4,
  })
); // mango: 17

  findBestEmployee({
    lux: 147,
    david: 21,
    kiwi: 19,
    chelsy: 38,
  })
); // lux: 147
