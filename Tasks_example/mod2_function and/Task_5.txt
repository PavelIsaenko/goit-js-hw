const add = function() {
  let args = Array.from(arguments);
  // console.log(args);

  const mult = args[0];
  args = args.slice(1);
  // console.log(args);

  let total = 0;
  for (let i = 0; i < args.length; i += 1) {
    total += args[i];
  }
  return total * mult;
};

console.log(add(5, 1, 2, 3));
console.log(add(10, 1, 2, 3, 4, 5));
console.log(add(15, 1, 2, 3, 4, 5, 6, 7, 8, 9));
